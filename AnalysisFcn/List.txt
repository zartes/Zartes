    c = distinguishable_colors(25,'w',func);

multiwaitbar.m
  MULTIWAITBAR displays multiple waitbars.
 
    Example:
    H = multiwaitbar(3,[0 0 0 0],{'Please wait','Please wait','Please wait'});
    for i = 0:0.2:1
      for j = 0:0.2:1
         for k = 0:0.1:1
           multiwaitbar(3,[i j k],{'Computing','Computing','Computing'},H);
         end
      end
    end
    delete(H.figure)
    clear('H')
